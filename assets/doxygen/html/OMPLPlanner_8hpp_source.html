51template <class PLANNER>
52class OMPLPlanner : public AbstractPlanner {
53 public:
54 OMPLPlanner() : AbstractPlanner("OMPL") {
55 _omplPlanner = ob::PlannerPtr(new PLANNER(ss->getSpaceInformation()));
56 AbstractPlanner::LastCreatedPlannerName = name();
57 }
58
59 ob::PlannerStatus run() override {
60 intermediarySolutions.clear();
61
62 ss->setPlanner(_omplPlanner);
63 ss->setup();
64
65 Stopwatch watch;
66 auto problem = _omplPlanner->getProblemDefinition();
67 problem->setIntermediateSolutionCallback(
68 [&](const ob::Planner *planner,
69 const std::vector<const ob::State *> &states, const ob::Cost cost) {
70 og::PathGeometric solution(global::settings.ompl.space_info);
71 // the OMPL intermediary solution doesn't include the start state
72 solution.append(global::settings.environment->startState());
73 for (const auto *state : states) solution.append(state);
74 // the OMPL intermediary solution doesn't include the goal state
75 solution.append(global::settings.environment->goalState());
76 IntermediarySolution is(watch.elapsed(), cost.value(), solution);
77 intermediarySolutions.emplace_back(is);
78 });
79
80 watch.start();
81 auto solved = ss->solve(global::settings.max_planning_time);
82 OMPL_INFORM("OMPL %s planning status: %s", _omplPlanner->getName().c_str(),
83 solved.asString().c_str());
84
85 if (solved) {
86 // Output the length of the path found
87 _solution.clear();
88 _solution.append(global::settings.environment->startState());
89 const auto &path = ss->getSolutionPath();
90 for (std::size_t i = 0; i < path.getStateCount(); ++i) {
91 _solution.append(path.getState(i));
92 }
93
94 OMPL_INFORM(
95 "%s found a solution of length %f with an optimization objective "
96 "value of %f",
97 _omplPlanner->getName().c_str(), _solution.length(),
98 _solution.cost(ss->getOptimizationObjective()));
99 } else
100 OMPL_WARN("No solution found.");
101 return solved;
102 }
103
104 std::string name() const override { return _omplPlanner->getName(); }
105
106 og::PathGeometric solution() const override {
107 og::PathGeometric path(_solution);
108 return og::PathGeometric(_solution);
109 }
110
111 bool hasReachedGoalExactly() const override {
112 return ss->haveExactSolutionPath();
113 }
114
115 double planningTime() const override {
116 return ss->getLastPlanComputationTime();
117 }
118
119 ob::Planner *omplPlanner() override { return _omplPlanner.get(); }
120
121 virtual nlohmann::json getSettings() const override {
122 nlohmann::json settings;
123 const auto &param_map = _omplPlanner->params().getParams();
124 for (const auto &[key, value_ptr] : param_map) {
125 settings[key] = value_ptr->getValue();
126 }
127 return settings;
128 }
129
130 private:
131 ob::PlannerPtr _omplPlanner;
132 og::PathGeometric _solution{global::settings.ompl.space_info};
133};
